async function load(q = '') { const token = 'הקוד_הארוך_ששלחת_לי_כאן'; const url = q ? `https://api.themoviedb.org/3/search/multi?query=${encodeURIComponent(q)}` : `https://api.themoviedb.org/3/trending/all/day`; try { const r = await fetch(url, { headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json;charset=utf-8' } }); const d = await r.json(); document.getElementById('g').innerHTML = d.results.filter(i=>i.poster_path).map(i => `

${i.title || i.name}

`).join(''); } catch (e) { console.error(e); } }
